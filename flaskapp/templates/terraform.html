Terraform

Минимальный пример VPC + VM (провайдер любой, подставьте свой):

terraform {
required_version = ">= 1.6"
required_providers {
yandex = {
source = "yandex-cloud/yandex"
version = ">= 0.123"
}
}
}


provider "yandex" {
zone = var.zone
}


resource "yandex_vpc_network" "net" {
name = "demo-net"
}


resource "yandex_vpc_subnet" "subnet" {
name = "demo-subnet"
zone = var.zone
network_id = yandex_vpc_network.net.id
v4_cidr_blocks = ["10.10.0.0/24"]
}


output "subnet_id" { value = yandex_vpc_subnet.subnet.id }

Команды: terraform fmt && terraform validate && terraform apply
